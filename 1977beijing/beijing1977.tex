\begin{questions}
	\question 解方程： \( \sqrt{x - 1} = 3 - x \).
	\begin{solution}
		\begin{align*}
			\sqrt{x - 1}  & = 3 - x        \\
			x - 1         & = 9 - 6x + x^2 \\
			x^2 - 7x + 10 & = 0            \\
			(x-2)(x-5)    & = 0            \\
			x_1 = 2, x_2 = 5
		\end{align*}
	\end{solution}

	\question 计算： \( 2^{-\frac12} + \frac{2^0}{\sqrt{2}} + \frac{1}{\sqrt{2} - 1}. \)

	\begin{solution}
		\begin{align*}
			\text{原式} & = \frac{1}{\sqrt{2}} + \frac{1}{\sqrt{2}} + \sqrt{2} + 1 \\
			            & = \frac{2}{\sqrt{2}} + \sqrt{2} + 1                      \\
			            & = 2\sqrt{2} + 1
		\end{align*}
	\end{solution}

\end{questions}
