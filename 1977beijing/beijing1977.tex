	\question 解方程： \( \sqrt{x - 1} = 3 - x \)
	\begin{solution}
		\begin{align*}
			\sqrt{x - 1}  & = 3 - x        \\
			x - 1         & = 9 - 6x + x^2 \\
			x^2 - 7x + 10 & = 0            \\
			(x-2)(x-5)    & = 0            \\
			x_1 = 2, x_2 = 5
		\end{align*}
	\end{solution}
